Run the following commands in order to set up a smartmirror

sudo apt-get update
sudo apt-get install python-pip python-dev ipython
sudo apt-get install bluetooth libbluetooth-dev
sudo apt-get install bluez bluez-util minicom
sudo pip install pybluez
sudo git clone https://github.com/LucasSpeer/SmartMirror.git
change line in /etc/systemd/system/dbus-org.bluez.service to ExecStart=/usr/lib/bluetooth/bluetoothd --compat
sudo chmod 777 /var/run/sdp
