Run the following commands in order to set up a smartmirror
 
This is mostly taken care of in firstBoot.sh and PIinit.sh


one time:

sudo apt-get update
sudo apt-get install ipython
sudo apt-get install libbluetooth-dev
sudo apt-get install apache2
sudo pip install pybluez
sudo git clone https://github.com/LucasSpeer/SmartMirror.git
sudo chmod 777 /var/run/sdp
cd ~/SmartMirror/lib/
sudo cp main.conf /etc/bluetooth/
sudo cp rc.local machine-info /etc/
sudo cp panel /home/pi/.config/lxpanel/LXDE-pi/panels/
sudo cp dbus-org.bluez.service /etc/systemd/system/
sudo hciconfig hci0 | grep Address | sudo nano addr  //creates te file addr.save with a line containing the local BT MAC addresshci
sudo python ~/SmartMirror/lib/rfcommset.py


Startup commands:
// hci0 up and piscan will eventually be tied to a button with a timeout
sudo hciconfig hci0 up
sudo hciconfig hci0 piscan
// these scripts start the bluetooth agent and the rfcomm server
sudo python ~/SmartMirror/lib/simple-agent &
sudo python ~/SmartMirror/lib/rfcomm-server.py &
